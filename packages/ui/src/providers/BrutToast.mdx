# Toast

Transient feedback via `BrutToastProvider` and the `useToast()` hook.

## Setup

```tsx
import { BrutToastProvider, useToast } from "@primo-brutality/ui";

function App() {
  return (
    <BrutToastProvider>
      <Screen />
    </BrutToastProvider>
  );
}

function Screen() {
  const { show } = useToast();
  return (
    <BrutButton onPress={() => show({ message: "Saved!", variant: "success" })}>
      Save
    </BrutButton>
  );
}
```
